function compare (a, b){
    let str_a = a.toString().split('.')[1].length;
    let str_b = b.toString().split('.')[1].length;
    return str_a > str_b ? str_a : str_b;
}

function addPoints(a, b){
    return Number((a + b).toFixed(compare(a, b)));
